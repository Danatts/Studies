fn main() {
    let mut foo: Vec<u32> = Vec::from([8, 1, 9, 2, 7, 4]);
    println!("{}", get_median(&mut foo));
}

fn get_median(array: &mut Vec<u32>) -> u32 {

    array.sort();
    
    let median = if array.len() % 2 != 0 {
        array[array.len() / 2]
    } else {
        (array[array.len() / 2] + array[(array.len() / 2) - 1]) / 2
    };

    println!("{:?}", array);
    median
}
